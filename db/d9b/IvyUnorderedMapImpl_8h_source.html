215 template __UMAPTPLARGSINIT__ __HOST_DEVICE__
216 IvyUnorderedMap __UMAPTPLARGS__::data_container IvyUnorderedMap __UMAPTPLARGS__::get_rehashed_data(IvyUnorderedMap __UMAPTPLARGS__::size_type new_n_buckets) const{
217 size_type const current_n_size_buckets = this->bucket_count();
218
219 size_type const preferred_data_capacity = hash_equal::preferred_data_capacity(new_n_buckets);
220 size_type const max_n_bucket_elements = (preferred_data_capacity+1)/new_n_buckets;
221
222 allocator_type a;
223 constexpr IvyMemoryType def_mem_type = IvyMemoryHelpers::get_execution_default_memory();
224 IvyMemoryType const mem_type = _data.get_memory_type();
225 IvyGPUStream* stream = _data.gpu_stream();
226 size_type const n_size = this->size();
227 size_type const n_capacity = this->capacity();
228
229 // Make a new data container to which we can copy the data with size = 0 and capacity = new_n_buckets.
230 data_container new_data = std_mem::make_unique<bucket_element>(
231 0, new_n_buckets,
232 mem_type, stream
233 );
234
235 {
236 bool const is_same_mem_type = (mem_type==def_mem_type);
237 build_GPU_stream_reference_from_pointer(stream, ref_stream);
238
239 memview<bucket_element, allocator_bucket_element> data_view(_data.get(), current_n_size_buckets, mem_type, &ref_stream, false);
240 {
241 bucket_element* data_ptr = data_view;
242 for (size_type ib=0; ib<current_n_size_buckets; ++ib){
243 auto& data_bucket = data_ptr->second;
244 size_type const n_size_data_bucket = data_bucket.size();
245 memview<value_type, allocator_type> data_bucket_view(data_bucket.get(), n_size_data_bucket, mem_type, &ref_stream, false);
246 {
247 value_type* data_bucket_ptr = data_bucket_view;
248 for (size_type jd=0; jd<n_size_data_bucket; ++jd){
249 auto const& data_value = *data_bucket_ptr;
250 auto const& key = data_value.first;
251 auto const& value = data_value.second;
252 auto const& hash = hasher()(key);
253
254 bool is_inserted = false;
255 size_type const new_current_n_size_buckets = new_data.size();
256 auto& d_new_data_ptr = new_data.get();
257 memview<bucket_element, allocator_bucket_element> new_data_view(d_new_data_ptr, new_current_n_size_buckets, mem_type, &ref_stream, false);
258 {
259 bucket_element* new_data_ptr = new_data_view;
260 for (size_type jb=0; jb<new_current_n_size_buckets; ++jb){
261 auto& new_bucket_element = *new_data_ptr;
262 auto const& new_bucket_hash = new_bucket_element.first;
263 if (hash_equal::eval(n_size, n_capacity, hash, new_bucket_hash)){
264 auto& new_data_bucket = new_bucket_element.second;
265 if (new_data_bucket.capacity()==0) new_data_bucket.reserve(max_n_bucket_elements, mem_type, stream);
266 new_data_bucket.emplace_back(key, value);
267 if (!is_same_mem_type){
268 auto tmp_tgt = d_new_data_ptr + jb;
269 IvyMemoryHelpers::transfer_memory(tmp_tgt, new_data_ptr, 1, mem_type, def_mem_type, ref_stream);
270 }
271 is_inserted = true;
272 break;
273 }
274 ++new_data_ptr;
275 }
276 if (!is_inserted){
277 // If we stil have not found the bucket, we need to create a new one.
278 new_data.emplace_back(
279 hash,
280 std_mem::build_unique<value_type>(
281 a, 1, max_n_bucket_elements,
282 mem_type, stream, data_value
283 )
284 );
285 }
286 }
287 ++data_bucket_ptr;
288 }
289 }
290 ++data_ptr;
291 }
292 }
293
294 destroy_GPU_stream_reference_from_pointer(stream);
295 }
296
297 return new_data;
298 }

313 template __UMAPTPLARGSINIT__ template<typename... Args>
314 __HOST_DEVICE__ void IvyUnorderedMap __UMAPTPLARGS__::insert_impl(
315 IvyMemoryType mem_type, IvyGPUStream* stream, Key const& key, Args&&... args
316 ){
317 size_type const current_size = this->size();
318 size_type const current_capacity = this->capacity();
319 size_type const current_bucket_size = this->bucket_count();
320 size_type const current_bucket_capacity = this->bucket_capacity();
321 size_type const preferred_current_data_capacity = hash_equal::preferred_data_capacity(current_bucket_capacity);
322 size_type const current_max_n_bucket_elements = (preferred_current_data_capacity+1)/current_bucket_capacity;
323
324 size_type const new_size = current_size + 1;
325 size_type const new_capacity = (new_size>current_capacity ? new_size + 1 : current_capacity);
326 size_type const new_bucket_capacity = hash_equal::bucket_size(new_size, new_capacity);
327 size_type const preferred_new_data_capacity = hash_equal::preferred_data_capacity(new_bucket_capacity);
328 size_type const new_max_n_bucket_elements = (preferred_new_data_capacity+1)/new_bucket_capacity;
329
330 constexpr IvyMemoryType def_mem_type = IvyMemoryHelpers::get_execution_default_memory();
331 hash_result_type const hash = hasher()(key);
332 allocator_type a;
333 bool trigger_it_reset = false;
334
335 if (!_data || _data.get_memory_type()!=mem_type){
336 _data = std_mem::make_unique<bucket_element>(
337 1, new_bucket_capacity,
338 mem_type, stream,
339 std_util::pair<hash_result_type, bucket_data_type>(
340 hash,
341 std_mem::build_unique<value_type>(
342 a, 1, new_max_n_bucket_elements,
343 mem_type, stream,
344 value_type(key, mapped_type(std_util::forward<Args>(args)...))
345 )
346 )
347 );
348 this->reset_iterator_builder();
349 }
350 else{
351 bool is_found = false;
352 value_type* mem_loc_pos = nullptr;
353 {
354 bool const is_same_mem_type = (mem_type==def_mem_type);
355 build_GPU_stream_reference_from_pointer(stream, ref_stream);
356
357 auto& data_ptr = _data.get();
358 memview<bucket_element, allocator_bucket_element> data_view(data_ptr, current_bucket_size, mem_type, &ref_stream, false);
359 bucket_element* tr_h_data_ptr = data_view;
360 for (size_type ib=0; ib<current_bucket_size; ++ib){
361 auto& current_bucket_element = *tr_h_data_ptr;
362 auto const& bucket_hash = current_bucket_element.first;
363 auto& data_bucket = current_bucket_element.second;
364 if (hash_equal::eval(current_size, current_capacity, hash, bucket_hash)){
365 auto& bucket_data_ptr = data_bucket.get();
366 size_type const n_size_data_bucket = data_bucket.size();
367 memview<value_type, allocator_type> bucket_data_view(bucket_data_ptr, n_size_data_bucket, mem_type, &ref_stream, false);
368 {
369 value_type* tr_h_bucket_data_ptr = bucket_data_view;
370 for (size_t jd=0; jd<n_size_data_bucket; ++jd){
371 value_type& data_value = *tr_h_bucket_data_ptr;
372 if (key_equal::eval(current_size, current_capacity, key, data_value.first)){
373 data_value.second = mapped_type(std_util::forward<Args>(args)...);
374 if (!is_same_mem_type){
375 auto tmp_tgt = bucket_data_ptr + jd;
376 IvyMemoryHelpers::transfer_memory(tmp_tgt, tr_h_bucket_data_ptr, 1, mem_type, def_mem_type, ref_stream);
377 }
378 is_found = true;
379 break;
380 }
381 ++tr_h_bucket_data_ptr;
382 }
383 }
384 if (!is_found){
385 trigger_it_reset = (data_bucket.size()==data_bucket.capacity());
386 data_bucket.emplace_back(key, mapped_type(std_util::forward<Args>(args)...));
387 mem_loc_pos = data_bucket.get() + data_bucket.size() - 1;
388 is_found = true;
389 }
390 }
391 if (is_found){
392 if (mem_type!=def_mem_type){
393 auto tmp_tgt = data_ptr + ib;
394 IvyMemoryHelpers::transfer_memory(tmp_tgt, tr_h_data_ptr, 1, mem_type, def_mem_type, ref_stream);
395 }
396 break;
397 }
398 ++tr_h_data_ptr;
399 }
400 if (!is_found){
401 // If we stil have not found the bucket, we need to create a new one.
402 _data.emplace_back(
403 hash,
404 std_mem::build_unique<value_type>(
405 a, 1, current_max_n_bucket_elements,
406 mem_type, stream, value_type(key, mapped_type(std_util::forward<Args>(args)...))
407 )
408 );
409 memview<bucket_element, allocator_bucket_element> data_view_last(_data.get() + current_bucket_size, mem_type, &ref_stream, false);
410 auto& data_bucket = (*data_view_last).second;
411 mem_loc_pos = data_bucket.get();
412 }
413
414 destroy_GPU_stream_reference_from_pointer(stream);
415 }
416
417 // We need to update the iterators if a new element is inserted.
418 // We do this even if we rehash later, which will invalidate the iterators.
419 // The reason is to make sure size() and capacity() are updated, which are need to be up-to-date in rehashing.
420 // Notice that the correct test for genuine new insertion is to check if mem_loc_pos is not a nullptr
421 // because it is also possible to find the inserted key to exist and just update the corresponding value.
422 if (mem_loc_pos){
423 // If the new size is greater than the current capacity, we need to reset the iterator builder;
424 // the push_back function does not invalidate existing iterators if the existing capacity of the iterator builder is exceeded.
425 // Otherwise, we just push the new element to its back.
426 if (new_size>current_capacity || trigger_it_reset) this->reset_iterator_builder();
427 else _iterator_builder.push_back(mem_loc_pos, mem_type, stream);
428 }
429 }

473 if (!_data) return iterator();
474 IvyUnorderedMap __UMAPTPLARGS__::iterator res = this->end();
475 constexpr IvyMemoryType def_mem_type = IvyMemoryHelpers::get_execution_default_memory();
476 IvyMemoryType const mem_type = _data.get_memory_type();
477 IvyGPUStream* stream = _data.gpu_stream();
478
479 // Note that current_size and current_capacity come from the size and capacity of the iterator builder.
480 // Since erase_impl does not update iterators, the actual values of the size and capacity of the _data container may be different.
481 // However, since erase_impl also does not launch rehashing, hash_equal::eval and key_equal::eval NEED current_size and current_capacity
482 // to correspond to the _data shape after the last rehashing operation.
483 // For that reason, we are using current_size and current_capacity correctly here.
484 size_type const current_size = this->size();
485 size_type const current_capacity = this->capacity();
486 size_type const current_bucket_size = this->bucket_count();
487
488 hash_result_type const hash = hasher()(key);
489 allocator_type a;
490 {
491 bool const is_same_mem_type = (mem_type==def_mem_type);
492 build_GPU_stream_reference_from_pointer(stream, ref_stream);
493
494 bucket_element*& data_ptr = _data.get();
495 memview<bucket_element, allocator_bucket_element> data_view(data_ptr, current_bucket_size, mem_type, &ref_stream, false);
496 {
497 bucket_element* tr_h_data_ptr = data_view.get() + current_bucket_size - 1;
498 for (size_type rib=0; rib<current_bucket_size; ++rib){
499 size_type const ib = current_bucket_size-1-rib;
500 auto& current_bucket_element = *tr_h_data_ptr;
501 auto const& bucket_hash = current_bucket_element.first;
502 auto& data_bucket = current_bucket_element.second;
503 size_type const n_size_data_bucket = data_bucket.size();
504 if (n_size_data_bucket>0 && hash_equal::eval(current_size, current_capacity, hash, bucket_hash)){
505 bool data_bucket_modified = false;
506
507 value_type*& bucket_data_ptr = data_bucket.get();
508 memview<value_type, allocator_type> bucket_data_view(bucket_data_ptr, n_size_data_bucket, mem_type, &ref_stream, false);
509 {
510 value_type* tr_h_bucket_data_ptr = bucket_data_view.get() + n_size_data_bucket - 1;
511 for (size_t rjd=0; rjd<n_size_data_bucket; ++rjd){
512 size_t const jd = n_size_data_bucket-1-rjd;
513 if (key_equal::eval(current_size, current_capacity, key, tr_h_bucket_data_ptr->first)){
514 data_bucket.erase(jd);
515 data_bucket_modified = true;
516 ++n_erased;
517 }
518 --tr_h_bucket_data_ptr;
519 }
520 }
521 if (data_bucket_modified && !is_same_mem_type){
522 auto tmp_tgt = data_ptr + ib;
523 IvyMemoryHelpers::transfer_memory(tmp_tgt, tr_h_data_ptr, 1, mem_type, def_mem_type, ref_stream);
524 }
525 }
526 if (data_bucket.size()==0) _data.erase(ib);
527 --tr_h_data_ptr;
528 }
529 }
530
531 destroy_GPU_stream_reference_from_pointer(stream);
532 }
