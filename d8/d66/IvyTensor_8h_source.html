228 template<typename T> struct IvyNodeSelfRelations<IvyTensor<T>>{
229 static constexpr bool is_conjugatable = is_conjugatable<T>;
230 static __HOST_DEVICE__ constexpr bool is_differentiable(IvyTensor<T> const& x){
231 if constexpr (is_constant_v<IvyTensor<T>>) return false;
232 bool res = false;
233 for (IvyTensorDim_t i=0; i<x.num_elements(); ++i){ res |= is_differentiable(x.data_[i]); if (res) break; }
234 return res;
235 }
236 static __HOST_DEVICE__ void conjugate(IvyTensor<T>& x){
237 if (!is_conjugatable) return;
238 IvyTensorDim_t const n = x.num_elements();
239 #define _CMD \
240 for (IvyTensorDim_t i=0; i<n; ++i){ conjugate(x.data_[i]); }
241#if defined(OPENMP_ENABLED)
242 if (n>=NUM_CPU_THREADS_THRESHOLD){
243 #pragma omp parallel for schedule(static)
244 _CMD
245 }
246 else
247#endif
248 {
249 _CMD
250 }
251 #undef _CMD
252 }

254 template<typename T, typename U> struct IvyNodeBinaryRelations<IvyTensor<T>, U>{
255 static __HOST_DEVICE__ bool depends_on(IvyTensor<T> const& fcn, U* var){
256 if (!var) return false;
257 bool res = std_mem::addressof(fcn) == var;
258 if (!res){
259 for (IvyTensorDim_t i=0; i<fcn.num_elements(); ++i){ res |= IvyMath::depends_on(fcn.at(i), var); if (res) break; }
260 }
261 return res;
262 }
