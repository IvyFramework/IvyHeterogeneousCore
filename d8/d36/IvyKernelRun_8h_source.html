100 IvyTypes::size_t ix = blockIdx.x * blockDim.x + threadIdx.x;
101 IvyTypes::size_t iy = blockIdx.y * blockDim.y + threadIdx.y;
102 IvyTypes::size_t iz = blockIdx.z * blockDim.z + threadIdx.z;
103 i = ix + iy * blockDim.x * gridDim.x + iz * blockDim.x * gridDim.x * blockDim.y * gridDim.y;

109 IvyTypes::size_t iz = blockIdx.z * blockDim.z + threadIdx.z;
110 j = iy + iz * blockDim.y * gridDim.y;
111 }
