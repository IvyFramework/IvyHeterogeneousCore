73 /*
74 reduced_value_type:
75 The value of reduced_value_t is supposed to be
76 the value_t type of the class if there is one, or an arithmetic type otherwise.
77 This means if
78 - T = arithmetic type, reduced_value_t = T.
79 - T = Ivy(Constant,'')Variable<U>, reduced_value_t = U (arithmetic type).
80 - T = IvyComplexVariable<U>, reduced_value_t = IvyComplexVariable<U>.
81 - T = IvyTensor<U>, reduced_value_t = IvyTensor<U>.
82 - T = IvyFunction<U>, reduced_value_t = IvyFunction<U>::value_t, which is an IvyVariable/ComplexVariable/Tensor<R>.
83 */
84 template<typename T> struct reduced_value_type{
85 template<typename U> static __HOST_DEVICE__ auto test_vtype(int) -> typename U::value_t;
86 template<typename U> static __HOST_DEVICE__ auto test_vtype(...) -> T;
87 using type = decltype(test_vtype<T>(0));
88 };
89 template<typename T> using reduced_value_t = typename reduced_value_type<T>::type;
90
91 /*
92 reduced_data_type:
93 The type reduced_data_t is supposed to be
94 the dtype_t type of the class if there is one, or an arithmetic type otherwise.
95 This means if
96 - T = arithmetic type, reduced_data_t = T.
97 - T = Ivy(Constant,'',Complex)Variable<U>, reduced_data_t = U (arithmetic type).
98 - T = IvyTensor<U>, reduced_data_t = U (which is not necessarily an arithmetic type).
99 - T = IvyFunction<U>, reduced_data_t = IvyFunction<U>::dtype_t.
100 */
101 template<typename T> struct reduced_data_type{
102 template<typename U> static __HOST_DEVICE__ auto test_dtype(int) -> typename U::dtype_t;
103 template<typename U> static __HOST_DEVICE__ auto test_dtype(...) -> T;
104 using type = decltype(test_dtype<T>(0));
105 };
106 template<typename T> using reduced_data_t = typename reduced_data_type<T>::type;

112 /*
113 fundamental_data_t:
114 The type fundamental_data_t is supposed to be the lowest baseline arithmetic type of each class.
115 Conceptually, this is different from reduced_data_t in that it is the lowest baseline type,
116 not necessarily the dtype_t of the data in IvyFunction or IvyTensor themselves.
117 In other words, fundamental_data_t looks for the dtype_t recursively in IvyFunction and IvyTensor
118 while reduced_data_t looks for the dtype_t only in the top level class.
119 It is specialized further for IvyFunction and IvyTensor later on.
120 To give a concrete example, if T = IvyFunction<IvyTensor<IvyVariable<double>>>,
121 - fundamental_data_t<T> = double, but
122 - reduced_data_t<T> = IvyTensor<IvyVariable<double>>.
123 */
124 template<typename T> struct fundamental_data_type{
125 using type = reduced_data_t<T>;
126 };
127 template<typename T> using fundamental_data_t = typename fundamental_data_type<T>::type;
