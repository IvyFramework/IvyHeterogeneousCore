85 __HOST_DEVICE__ memview(pointer const& ptr, IvyMemoryType owning_mem_type_, IvyGPUStream* stream_, bool recursive_, bool force_transfer_=false) :
86 data(nullptr), s(1), n(1), stream(stream_), recursive(recursive_)
87 {
88 constexpr IvyMemoryType def_mem_type = IvyMemoryHelpers::get_execution_default_memory();
89 do_own = (def_mem_type != owning_mem_type_ || force_transfer_);
90 if (!do_own){
91 data = ptr;
92 return;
93 }
94 build_GPU_stream_reference_from_pointer(stream, ref_stream);
95 if (!recursive){
96 // In the non-recursive case, only the bytes in memory are copied. The contents of the object remain as in the original object.
97 // For that reason, the accompanying statement to free the data pointer would be simply IvyMemoryHelpers::free_memory.
98 IvyMemoryHelpers::allocate_memory(data, n, def_mem_type, ref_stream);
99 IvyMemoryHelpers::transfer_memory(data, ptr, s, def_mem_type, owning_mem_type_, ref_stream);
100 }
101 else{
102 // In the recursive case, the object is copied as well.
103 // The accompanying statement to free the data pointer would be allocator_traits_t::destroy
104 // because new internal objects are created and need to be destroyed as well.
105 allocator_traits_t::allocate(data, n, def_mem_type, ref_stream);
106 allocator_traits_t::transfer(data, ptr, s, def_mem_type, owning_mem_type_, ref_stream);
107 }
