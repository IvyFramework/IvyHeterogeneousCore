12 IvyNodeSelfRelations:
13 Any node has dependence and differentiability properties.
14 Because CUDA does not support RTTI, we need to implement these properties in a separate class and check them through static 'polymorphism'.
15 The idea is to provide partial specializations when needed in order to implement these properties.
16
17 IvyNodeSelfRelations::is_differentiable: Check if this node is differentiable. By default, no node is differentiable.
18 IvyNodeSelfRelations::is_conjugatable: By default, no node is conjugatable.
19 */
20 template<typename T> struct IvyNodeSelfRelations{
21 static constexpr bool is_conjugatable = false;
22 static __HOST_DEVICE__ void conjugate(T&){}
23 static __HOST_DEVICE__ constexpr bool is_differentiable(T const&){ return false; }
24 };

26 is_conjugatable: Convenience function to check if a node is conjugatable.
27 */
28 template<typename T> inline constexpr bool is_conjugatable = IvyNodeSelfRelations<T>::is_conjugatable;

30 conjugate: Convenience function to conjugate a node.

32 template<typename T> __HOST_DEVICE__ void conjugate(T& x){ IvyNodeSelfRelations<T>::conjugate(x); }

34 is_differentiable: Convenience function to check if a node is differentiable.
