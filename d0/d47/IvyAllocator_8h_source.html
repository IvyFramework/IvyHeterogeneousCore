615 T*& target, U* const& source,
616 size_t n_tgt_init, size_t n_tgt, size_t n_src
617 , IvyMemoryType type_tgt, IvyMemoryType type_src
618 , IvyGPUStream& stream
619 ){
620 bool res = true;
621#if (DEVICE_CODE == DEVICE_CODE_HOST) && defined(__USE_CUDA__)
622 bool const tgt_on_device = use_device_acc(type_tgt);
623 bool const src_on_device = use_device_acc(type_src);
624#else
625 constexpr bool tgt_on_device = true;
626 constexpr bool src_on_device = true;
627#endif
628 if (n_tgt==0 || n_src==0 || !source) return false;
629 if (!(n_src==n_tgt || n_src==1)){
630#if COMPILER == COMPILER_MSVC
631 __PRINT_ERROR__("IvyMemoryHelpers::copy_data: Invalid values for n_tgt=%Iu, n_src=%Iu\n", n_tgt, n_src);
632#else
633 __PRINT_ERROR__("IvyMemoryHelpers::copy_data: Invalid values for n_tgt=%zu, n_src=%zu\n", n_tgt, n_src);
634#endif
635 assert(0);
636 }
637 if (n_tgt_init!=n_tgt){
638 res &= std_ivy::deallocator_primitive<T>::destroy(target, n_tgt_init, type_tgt, stream);
639 res &= std_ivy::allocator_primitive<T>::allocate(target, n_tgt, type_tgt, stream);
640 }
641 if (res){
642#ifdef __USE_CUDA__
643 if (!src_on_device && !tgt_on_device) res = false;
644 else{
645 U* d_source = (src_on_device ? source : nullptr);
646 if (!src_on_device){
647 res &= std_ivy::allocator_primitive<U>::allocate(d_source, n_src, IvyMemoryType::GPU, stream);
648 res &= std_ivy::transfer_memory_primitive<U>::transfer(d_source, source, n_src, IvyMemoryType::GPU, type_src, stream);
649 }

651 if (!tgt_on_device) res &= std_ivy::allocator_primitive<T>::allocate(d_target, n_tgt, IvyMemoryType::GPU, stream);
652 else d_target = target;
653 res &= run_kernel<copy_data_kernel<T, U>>(0, stream).parallel_1D(n_tgt, n_src, d_target, d_source);
654 if (!tgt_on_device){
655 res &= std_ivy::transfer_memory_primitive<T>::transfer(target, d_target, n_tgt, type_tgt, IvyMemoryType::GPU, stream);
656 res &= std_ivy::deallocator_primitive<T>::deallocate(d_target, n_tgt, IvyMemoryType::GPU, stream);
657 }
658 if (!src_on_device) res &= std_ivy::deallocator_primitive<U>::deallocate(d_source, n_src, IvyMemoryType::GPU, stream);
659 }
660 if (!res){
661 if (tgt_on_device!=src_on_device){
662 __PRINT_ERROR__("IvyMemoryHelpers::copy_data: Failed to copy data between host and device.\n");
663 assert(0);
664 }
