74 template<typename C, typename T>
75 __INLINE_FCN_RELAXED__ __HOST_DEVICE__ T op_parallel(
76 T* h_vals, IvyTypes::size_t n, IvyTypes::size_t n_serial,
77 IvyMemoryType mem_type_vals, IvyGPUStream& stream, int dyn_shared_mem = 0
78 ){
79 auto obj_allocator = std_mem::allocator<T>();
80
81 IvyTypes::size_t neff = 0;
82 C::parallel_op_n_mem(n, n_serial, neff);
83
84 auto h_res = obj_allocator.allocate(1, mem_type_vals, stream);
85 auto d_vals = obj_allocator.allocate(neff, IvyMemoryType::GPU, stream);
86
87 obj_allocator.transfer(d_vals, h_vals, n, IvyMemoryType::GPU, mem_type_vals, stream);
88 op_parallel_core<C, T>(n, n_serial, d_vals, stream, dyn_shared_mem);
89 obj_allocator.transfer(h_res, (d_vals+(neff-1)), 1, mem_type_vals, IvyMemoryType::GPU, stream);
90
91 // Stream synchronization is needed because h_res in host code might otherwise be deallocated before the transfer finishes.
92#ifndef __CUDA_DEVICE_CODE__
93 stream.synchronize();
94#endif
95
96 obj_allocator.deallocate(d_vals, neff, IvyMemoryType::GPU, stream);
97
98 T res = *h_res;
99 obj_allocator.deallocate(h_res, 1, mem_type_vals, stream);
100
101 return res;
102 }
